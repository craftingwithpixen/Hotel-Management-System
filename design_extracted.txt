1. Project Overview
HospitalityOS is a unified hotel and restaurant management platform built on the MERN stack. It covers every operational surface — front desk, kitchen, restaurant floor, inventory, HR, billing, and the customer-facing booking experience — in a single codebase. All features are interconnected through a shared MongoDB database and a real-time Socket.io event bus. Role-based access control gates every route, both on the server and in the React client.

2.3 Middleware Enforcement
authenticate.js
// middleware/authenticate.js
const authenticate = async (req, res, next) => {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.status(401).json({ message: "No token" });
  try {
    const decoded = jwt.verify(token, process.env.ACCESS_TOKEN_SECRET);
    req.user = await User.findById(decoded.id).select("-password -refreshToken");
    next();
  } catch (err) {
    return res.status(401).json({ message: "Token expired or invalid" });
  }
};

authorize.js
// middleware/authorize.js
const authorize = (...roles) => (req, res, next) => {
  if (!roles.includes(req.user.role))
    return res.status(403).json({ message: "Access denied" });
  next();
};

// Usage on a route:
router.delete("/staff/:id", authenticate, authorize("admin"), staffCtrl.delete);

// sockets/socketHandler.js
module.exports = (io) => {
  io.on("connection", (socket) => {
    // Verify JWT from handshake auth
    const token = socket.handshake.auth.token;
    const user = verifyAccessToken(token); // throws on invalid

    socket.on("join:kitchen", () => {
      if (["chef", "admin"].includes(user.role))
        socket.join("kitchen");
    });

    socket.on("join:table", ({ tableId }) => {
      socket.join(`table:${tableId}`);
    });

    socket.on("disconnect", () => {
      // Socket.io auto-removes from rooms
    });
  });
};

// services/socketService.js — emit helpers
const emitNewOrder = (io, order) => {

    orderId: order._id,
    tableNumber: order.table.tableNumber,
    items: order.items,
    waiterName: order.waiter.name,
  });
};

7.5 Low-Stock Cron Alert Flow
// jobs/stockAlertJob.js
const cron = require("node-cron");

cron.schedule("0 */6 * * *", async () => {
  const lowItems = await InventoryItem.find({
    $expr: { $lte: ["$currentStock", "$lowStockThreshold"] },
    isDeleted: false,
  }).populate("hotel supplier");

  if (lowItems.length === 0) return;

  // Group by hotel
  const byHotel = groupBy(lowItems, item => item.hotel._id.toString());

  for (const [hotelId, items] of Object.entries(byHotel)) {
    // 1. Emit socket event to admin room
    io.to("admin").emit("inventory:alert", {
      items: items.map(i => ({ name: i.name, currentStock: i.currentStock,
                               threshold: i.lowStockThreshold, unit: i.unit })),
    });

    // 2. Send email via Resend
    await emailService.sendLowStockAlert(items[0].hotel.email, items);
  }
});

8. Third-Party Integration Details
8.1 Razorpay
Server creates a Razorpay order (amount in paise). Client renders the Razorpay checkout using their script. After payment, the client sends the three Razorpay identifiers to /payments/verify. The server validates the HMAC-SHA256 signature using the Razorpay key secret before confirming anything.
// config/razorpay.js
const Razorpay = require("razorpay");
module.exports = new Razorpay({
  key_id: process.env.RAZORPAY_KEY_ID,
  key_secret: process.env.RAZORPAY_KEY_SECRET,
});

// utils/razorpayVerify.js
const crypto = require("crypto");
const verifyPayment = (orderId, paymentId, signature) => {
  const body = orderId + "|" + paymentId;
  const expected = crypto
    .createHmac("sha256", process.env.RAZORPAY_KEY_SECRET)
    .update(body)
    .digest("hex");
  return expected === signature;
};

// Webhook: POST /payments/webhook
// Use express.raw() (not express.json()) for this route to verify signature
const webhookSecret = process.env.RAZORPAY_WEBHOOK_SECRET;
const body = req.body.toString();
const signature = req.headers["x-razorpay-signature"];
const digest = crypto.createHmac("sha256", webhookSecret).update(body).digest("hex");
if (digest !== signature) return res.status(400).send("Invalid webhook");

8.2 Cloudinary
Images are uploaded client-side directly to Cloudinary using a signed upload signature generated by the backend. The server never handles raw file streams, keeping memory usage low.
// config/cloudinary.js
const cloudinary = require("cloudinary").v2;
cloudinary.config({
  cloud_name: process.env.CLOUDINARY_CLOUD_NAME,
  api_key: process.env.CLOUDINARY_API_KEY,
  api_secret: process.env.CLOUDINARY_API_SECRET,
});

// Generate signed upload params (called by client before upload)
const getSignedUploadParams = (folder, publicId) => {
  const timestamp = Math.round(new Date().getTime() / 1000);
  const signature = cloudinary.utils.api_sign_request(
    { timestamp, folder, public_id: publicId },
    process.env.CLOUDINARY_API_SECRET
  );
  return { timestamp, signature, folder, publicId,
           apiKey: process.env.CLOUDINARY_API_KEY };
};

8.5 QR Code Table Ordering
// services/qrService.js
const QRCode = require("qrcode");

const generateTableQR = async (tableId) => {
  const url = `${process.env.FRONTEND_URL}/scan/table/${tableId}`;
  const pngBuffer = await QRCode.toBuffer(url, { type: "png", width: 400 });

  // Upload to Cloudinary
  const result = await new Promise((res, rej) => {
    cloudinary.uploader.upload_stream(
      { folder: "qr_codes", public_id: `table_${tableId}` },
      (err, data) => err ? rej(err) : res(data)
    )(pngBuffer);
  });

  return { qrCode: result.secure_url, qrUrl: url };
};
